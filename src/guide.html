WHITE и BLACK — цвета, как краска для тачки. FPS = 60 — это чтобы игра не лагала, как мой старый комп на Windows 98!

Создание класса машины

Ща замутим тачку, пацаны! Класс Car — это как чертёж для твоей "Тойоты" или "Ниссана". Будет дрифтовать, как в Initial D!

Шаг 1: Определите класс Car

Вбей код для создания тачки:

class Car(pygame.sprite.Sprite):
    def __init__(self, x, y, image_path):
        super().__init__()
        # Проверка наличия файла изображения
        if not os.path.exists(image_path):
            raise FileNotFoundError(f"Файл {image_path} не найден")
        self.original_image = pygame.image.load(image_path).convert_alpha()
        self.original_image = pygame.transform.scale(self.original_image, (37, 75))
        self.image = self.original_image
        self.rect = self.image.get_rect(center=(x, y))
        
        self.speed = 0
        self.max_speed = 10
        self.acceleration = 0.15
        self.deceleration = 0.15 / 10
        self.angle = 0
        self.rotation_speed = 5 / 2

Глад объясняет: Смотри, class Car(pygame.sprite.Sprite) — это как сказать: "Эй, тачка, ты теперь спрайт, будешь красиво ездить!" __init__ — это где мы задаём координаты x, y и картинку image_path. Проверяем, есть ли файл, а то без тачки не погоняешь! pygame.image.load загружает картинку, transform.scale делает её компактной, как мой старый гараж. speed, acceleration, angle — это как газ, тормоз и руль, всё для дрифта!

Шаг 2: Реализуйте метод update

Теперь добавим управление, чтобы тачка гоняла:

    def update(self, keys, control_type):
        # Изменение скорости (ускорение и торможение)
        if control_type == "wasd":
            if keys[pygame.K_w]:
                self.speed = min(self.speed + self.acceleration, self.max_speed)
            elif keys[pygame.K_s]:
                self.speed = max(self.speed - self.acceleration, -self.max_speed / 2)
            else:
                if self.speed > 0:
                    self.speed = max(self.speed - self.deceleration, 0)
                elif self.speed < 0:
                    self.speed = min(self.speed + self.deceleration, 0)
        
        elif control_type == "arrows":
            if keys[pygame.K_UP]:
                self.speed = min(self.speed + self.acceleration, self.max_speed)
            elif keys[pygame.K_DOWN]:
                self.speed = max(self.speed - self.acceleration, -self.max_speed / 2)
            else:
                if self.speed > 0:
                    self.speed = max(self.speed - self.deceleration, 0)
                elif self.speed < 0:
                    self.speed = min(self.speed + self.deceleration, 0)

        # Поворот (разрешён всегда)
        if control_type == "wasd":
            if keys[pygame.K_a]:
                self.angle += self.rotation_speed
            if keys[pygame.K_d]:
                self.angle -= self.rotation_speed
        elif control_type == "arrows":
            if keys[pygame.K_LEFT]:
                self.angle += self.rotation_speed
            if keys[pygame.K_RIGHT]:
                self.angle -= self.rotation_speed
        
        # Обновление положения на основе скорости и угла
        if abs(self.speed) > 0:
            self.rect.x += self.speed * math.cos(math.radians(self.angle + 90))
            self.rect.y -= self.speed * math.sin(math.radians(self.angle + 90))

        # Вращение изображения
        self.image = pygame.transform.rotate(self.original_image, self.angle)
        self.rect = self.image.get_rect(center=self.rect.center)
        
        # Ограничение по экрану
        self.rect.clamp_ip(screen.get_rect())

Глад объясняет: Ох, пацаны, тут движуха! update — это как руль и педали в одном. Если жмёшь W или стрелку вверх — тачка разгоняется, S или вниз — тормозит. A, D или стрелки влево-вправо крутят руль. math.cos и math.sin — это как рассчитать, куда тачка поедет, прям как я в молодости, когда пытался в физику на уроках! transform.rotate вертит картинку, чтоб тачка красиво дрифтовала, а clamp_ip не даёт свалить за экран, как я однажды с трассы на мопеде!

Забавный факт от Глада: В Need for Speed: Underground неон под тачками был такой модный, что я чуть под свой Жигуль лампочки не прикрутил! Всё из японской уличной движухи 2000-х!

Создание игрового цикла

Пацаны, игровой цикл — это как сердце движка, бьётся и всё крутит! Без него игра — как тачка без бензина.

Шаг 1: Загрузка трассы и создание машин

Врубай функцию game_loop и грузи трассу с тачками:

def game_loop():
    # Проверка наличия файла трассы
    track_path = "track.png"
    if not os.path.exists(track_path):
        raise FileNotFoundError(f"Файл {track_path} не найден")
    
    # Загружаем изображение трассы
    track_image = pygame.image.load(track_path).convert_alpha()
    track_image = pygame.transform.scale(track_image, (SCREEN_WIDTH, SCREEN_HEIGHT))

    # Создание машинок
    try:
        car1 = Car(SCREEN_WIDTH // 8, SCREEN_HEIGHT // 2, "1.png")
        car2 = Car(SCREEN_WIDTH // 6, SCREEN_HEIGHT // 2, "9.png")
    except FileNotFoundError as e:
        print(e)
        pygame.quit()
        return

    all_sprites = pygame.sprite.Group()
    all_sprites.add(car1, car2)

Глад объясняет: Тут мы грузим трассу track.png — это как асфальт на твоей улице. Проверяем, что файл есть, а то без трассы — как без пельменей на ужин! Car создаёт две тачки, ставим их рядом, как на старте в Initial D. all_sprites — это как гараж, где все тачки в куче, чтобы их разом рисовать.

Шаг 2: Добавьте шрифт и основной цикл

Кидаем шрифт для скорости и главный цикл:

    # Инициализация шрифта с поддержкой кириллицы
    try:
        font = pygame.font.SysFont("arial", 36)
    except:
        font = pygame.font.Font(None, 36)

    running = True
    while running:
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                running = False
        
        keys = pygame.key.get_pressed()
        
        # Обновляем положение машинок
        car1.update(keys, "wasd")
        car2.update(keys, "arrows")
        
        # Отрисовываем трассу на экране
        screen.blit(track_image, (0, 0))
        
        # Отрисовываем машинки
        all_sprites.draw(screen)

        # Отображение скорости первой машинки
        speed_text1 = f"Скорость 1: {car1.speed:.2f}"
        text_surface1 = font.render(speed_text1, True, BLACK)
        screen.blit(text_surface1, (10, 10))

        # Отображение скорости второй машинки
        speed_text2 = f"Скорость 2: {car2.speed:.2f}"
        text_surface2 = font.render(speed_text2, True, BLACK)
        screen.blit(text_surface2, (SCREEN_WIDTH - text_surface2.get_width() - 10, 10))
        
        pygame.display.flip()
        clock.tick(FPS)

Глад объясняет: font — это как наклейка со спидометром, показывает, как быстро ты гонишь. while running — это как крутить педали без остановки. Проверяем, не закрыл ли ты окно (pygame.QUIT), а то всё, гонка окончена! keys ловит, что ты жмёшь, и обновляет тачки. screen.blit рисует трассу и тачки, а pygame.display.flip — как сказать: "Эй, комп, покажи картинку!" clock.tick(FPS) держит 60 кадров, чтоб не лагало, как мой старый видик!

Шаг 3: Запустите цикл

Врубай игруху:

if __name__ == "__main__":
    try:
        game_loop()
    except Exception as e:
        print(f"Произошла ошибка: {e}")
        pygame.quit()

Глад объясняет: if __name__ == "__main__" — это как ключ зажигания, запускает game_loop. try-except ловит косяки, если что-то пошло не так, как когда я пытался движок от Жигуля на мопед прикрутить!

Улучшения и доработки

Хочешь сделать игруху ещё круче? Вот пара идей, как тюнинговать тачку!

  • Столкновения: Кидай pygame.sprite.collide_rect(car1, car2), чтоб тачки бились, как в настоящей гонке!
  • Физика: Добавь трение или дрифт, чтоб как в Initial D!
  • Трасса: Врубай pygame.mask.from_surface для коллизий с дорогой.
  • Счётчик кругов: Замути контрольные точки, как на ралли!
  • Графика: Звуки мотора, дым, искры — как неон на моей тачке в 90-х!

Пример для столкновений:

if pygame.sprite.collide_rect(car1, car2):
    car1.speed *= 0.5
    car2.speed *= 0.5

Глад объясняет: collide_rect — это как бампер в бампер! Если тачки стукнулись, скорость падает, как мой энтузиазм, когда пельмени кончились. Остальные идеи — это как прикрутить спойлер или заниженную подвеску, делай и будешь король трассы!

Забавный факт от Глада: В F-Zero на SNES был Mode 7, делал трассы как 3D, хотя всё плоское! Это как я в 90-х пытался в зеркале выглядеть моложе!

Итоговый код

Вот тебе, пацан, весь код в одном флаконе! Это как полностью собранная тачка, готова к заезду. Кидай в main.py, грузи картинки и врубай движок!

import pygame
import sys
import math
import os

# Инициализация Pygame
pygame.init()

# Настройки окна
SCREEN_WIDTH = 1800
SCREEN_HEIGHT = 1000
screen = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT))
pygame.display.set_caption("Гонки")

# Цвета
WHITE = (255, 255, 255)
BLACK = (0, 0, 0)

# Частота кадров
FPS = 60
clock = pygame.time.Clock()

# Класс для машинки
class Car(pygame.sprite.Sprite):
    def __init__(self, x, y, image_path):
        super().__init__()
        # Проверка наличия файла изображения

        self.original_image = pygame.image.load(image_path).convert_alpha()
        self.original_image = pygame.transform.scale(self.original_image, (37, 75))
        self.image = self.original_image
        self.rect = self.image.get_rect(center=(x, y))
        
        self.speed = 0
        self.max_speed = 10
        self.acceleration = 0.15   # Увеличено для более отзывчивого управления
        self.deceleration = 0.15 / 10  # Увеличено для более плавного торможения
        self.angle = 0 
        self.rotation_speed = 5 / 2

        if control_type == "wasd":
            if keys[pygame.K_w]:
                self.speed = min(self.speed + self.acceleration, self.max_speed)
            elif keys[pygame.K_s]:
                self.speed = max(self.speed - self.acceleration, -self.max_speed / 2)
            else:

                    self.speed = max(self.speed - self.deceleration, 0)
                elif self.speed < 0:
                    self.speed = min(self.speed + self.deceleration, 0)
        
        elif control_type == "arrows":
            if keys[pygame.K_UP]:
                self.speed = min(self.speed + self.acceleration, self.max_speed)
            elif keys[pygame.K_DOWN]:
                self.speed = max(self.speed - self.acceleration, -self.max_speed / 2)
            else:
                if self.speed > 0:
                    self.speed = max(self.speed - self.deceleration, 0)
                elif self.speed < 0:
                    self.speed = min(self.speed + self.deceleration, 0)

        # Поворот (разрешён всегда)
        if control_type == "wasd":
            if keys[pygame.K_a]:
                self.angle += self.rotation_speed
            if keys[pygame.K_d]:

def game_loop():
    # Проверка наличия файла трассы
    track_path = "track.png"
    if not os.path.exists(track_path):
        raise FileNotFoundError(f"Файл {track_path} не найден")
    
    # Загружаем изображение трассы
    track_image = pygame.image.load(track_path).convert_alpha()
    track_image = pygame.transform.scale(track_image, (SCREEN_WIDTH, SCREEN_HEIGHT))

    # Создание машинок
    try:
        car1 = Car(SCREEN_WIDTH // 8, SCREEN_HEIGHT // 2, "1.png")
        car2 = Car(SCREEN_WIDTH // 6, SCREEN_HEIGHT // 2, "9.png")
    except FileNotFoundError as e:
        print(e)
        pygame.quit()
        return

    all_sprites = pygame.sprite.Group()
    all_sprites.add(car1, car2)

    # Инициализация шрифта с поддержкой кириллицы
    try:
        font = pygame.font.SysFont("arial", 36)  # Arial обычно поддерживает кириллицу
    except:
        font = pygame.font.Font(None, 36)  # Запасной вариант

    running = True
    while running:
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                running = False
        
        keys = pygame.key.get_pressed()
        
        # Обновляем положение машинок
        car1.update(keys, "wasd")
        car2.update(keys, "arrows")
        
        # Отрисовываем трассу на экране
        screen.blit(track_image, (0, 0))
        
        # Отрисовываем машинки
        all_sprites.draw(screen)

        # Отображение скорости первой машинки
        speed_text1 = f"Скорость 1: {car1.speed:.2f}"
        text_surface1 = font.render(speed_text1, True, BLACK)
        screen.blit(text_surface1, (10, 10))

        # Отображение скорости второй машинки
        speed_text2 = f"Скорость 2: {car2.speed:.2f}"
        text_surface2 = font.render(speed_text2, True, BLACK)
        screen.blit(text_surface2, (SCREEN_WIDTH - text_surface2.get_width() - 10, 10))
        
        pygame.display.flip()
        clock.tick(FPS)  # Ограничение FPS

    pygame.quit()

if __name__ == "__main__":
    try:
        game_loop()
    except Exception as e:
        print(f"Произошла ошибка: {e}")
        pygame.quit()
